Signals — the new reactivity model

Signals are synchronous, pull-based reactive primitives. Angular tracks which template reads which signal and schedules re-renders only for components where a read signal has changed. No Zone.js dirty-checking the entire tree.

Before — RxJS BehaviorSubject

count$ = new BehaviorSubject(0);

increment() {
  this.count$.next(this.count$.value + 1);
}

// template
{{ count$ | async }}

After — Angular 16+ signals

count   = signal(0);
doubled = computed(() => this.count() * 2);

increment() {
  this.count.update(v => v + 1);
}

// template — no async pipe needed
{{ count() }}  {{ doubled() }}

Signals integrate natively with OnPush. A signal read in a template marks that component for check automatically when the signal changes — no markForCheck() required.

Signal inputs & outputs (v17.1+)

// Signal inputs — replaces @Input() decorator
title  = input.required<string>();           // required, throws if missing
count  = input(0);                            // optional with default
alias  = input<string>('', { alias: 'lbl' }); // with alias

// Derive from inputs using computed()
upper = computed(() => this.title().toUpperCase());

// Signal output — replaces @Output() + EventEmitter
clicked = output<MouseEvent>();
// template: (click)="clicked.emit($event)"

// Route params auto-bound to signal inputs (requires withComponentInputBinding())
id = input.required<string>();  // bound from :id route param

RxJS interop

// RxJS Observable → Signal (for template consumption)
private readonly users$ = this.http.get<User[]>('/api/users').pipe(shareReplay(1));
readonly users = toSignal(this.users$, { initialValue: [] });

// Signal → Observable (to feed into RxJS pipelines)
readonly userId$ = toObservable(this.selectedUserId);
readonly user$   = this.userId$.pipe(
  switchMap(id => this.http.get<User>(`/users/${id}`))
);

Standalone architecture

Standalone components own their own imports array. No NgModule required. The app bootstraps from a component + provider configuration object.

// main.ts — standalone bootstrap (no AppModule)
bootstrapApplication(AppComponent, {
  providers: [
    provideRouter(routes, withComponentInputBinding(), withPreloading(PreloadAllModules)),
    provideHttpClient(withInterceptorsFromDi()),
    provideAnimationsAsync(),
    provideClientHydration(),  // SSR hydration (v17+)
  ]
});

// feature.component.ts — imports are per-component
@Component({
  standalone: true,
  selector: 'app-feature',
  imports: [RouterOutlet, NgOptimizedImage, UserCardComponent, ReactiveFormsModule],
  template: `
    @if (user()) { <user-card [user]="user()" /> }
    <router-outlet />
  `
})
export class FeatureComponent {
  user = toSignal(inject(UserService).getUser$());
}

// app.routes.ts
export const routes: Routes = [

Dependency injection — modern patterns

inject() function

Replaces constructor injection. Works in field initializers, factory functions, guards, resolvers, interceptors — anywhere inside an injection context.

DestroyRef

Register cleanup callbacks without implementing OnDestroy. Works in services and directives too. Combine with takeUntilDestroyed() for RxJS auto-unsubscription.

InjectionToken

Type-safe DI for non-class values (config objects, feature flags, env vars). Use providedIn: 'root' with a factory for tree-shakeable tokens.

EnvironmentProviders

Scope providers to a route subtree via providers: in route config. Replaces lazy-loaded module providers entirely.

@Component({ standalone: true, ... })
export class UserComponent {
  // inject() in field initializers — no constructor needed
  private http   = inject(HttpClient);
  private router = inject(Router);
  private ref    = inject(DestroyRef);
  private config = inject(APP_CONFIG);   // InjectionToken

  // Automatic cleanup when component is destroyed
  users$ = this.http.get<User[]>('/api/users').pipe(
    takeUntilDestroyed(this.ref)
  );
}

// InjectionToken — tree-shakeable
export const APP_CONFIG = new InjectionToken<AppConfig>('app.config', {
  providedIn: 'root',
  factory: () => ({ apiUrl: 'https://api.example.com', debug: false })
});

// Functional HTTP interceptor (v15+) — uses inject() internally
export const authInterceptor: HttpInterceptorFn = (req, next) => {
  const token = inject(AuthService).token();    // signal!
  return next(
    req.clone({ setHeaders: { Authorization: `Bearer ${token()}` } })
  );
};

Change detection

// OnPush — recommended for all components
@Component({
  changeDetection: ChangeDetectionStrategy.OnPush,
  ...
})
export class UserListComponent {
  // These trigger re-render automatically:
  users     = input.required<User[]>();            // @Input ref change
  userCount = computed(() => this.users().length); // signal-derived
  users$    = toSignal(this.userService.users$);   // async pipe equivalent

  // Escape hatch — only when mutating non-signal state
  constructor(private cdr: ChangeDetectorRef) {}
  refresh() { this.cdr.markForCheck(); }
}

// Zoneless — Angular 18 experimental
bootstrapApplication(AppComponent, {
  providers: [provideExperimentalZonelessChangeDetection()]
});

New control flow & @defer (v17)

Old structural directives

*ngIf="user; else loading"
*ngFor="let u of users; trackBy: trackById"
[ngSwitch]="role"
  *ngSwitchCase="'admin'"
  *ngSwitchDefault

New built-in control flow

@if (user) { ... } @else { <spinner/> }
@for (u of users; track u.id) { ... }
@empty { <p>No users</p> }
@switch (role) {
  @case ('admin') { <admin-panel/> }
  @default { <user-panel/> }
}

@for requires a track expression — not optional like trackBy. Always use a unique identifier. Using track $index disables list reconciliation and should only be a last resort.

@defer — lazy loading with declarative triggers

@defer (on viewport; prefetch on idle) {
  <heavy-analytics-chart [data]="chartData" />
} @placeholder (minimum 400ms) {
  <div class="chart-skeleton" aria-busy="true"></div>
} @loading (minimum 200ms) {
  <loading-spinner size="sm" />
} @error {
  <p>Chart failed to load. <a (click)="retry()">Retry</a></p>
}

// Interaction trigger — load when user clicks a button
<button #loadTrigger>Load reviews</button>
@defer (on interaction(loadTrigger)) { <reviews /> }

// Conditional trigger — load when condition becomes true
@defer (when userIsAuthenticated()) { <private-content /> }

prefetch on idle downloads the bundle during browser idle time — so when the trigger fires, the component loads from cache (near-instant). Zero bundle cost until needed.

Performance patterns

NgOptimizedImage

Automatic <link rel="preload"> for priority images, responsive srcset generation, lazy loading by default, CDN integration for format negotiation (WebP/AVIF).

@defer on viewport

Below-fold components defer rendering until they scroll into view. Bundle downloaded only when needed. Combine with prefetch on idle for instant-feel loading.

SSR + hydration (v17)

Full app hydration enabled by default. DOM reuse from server render prevents layout shift. ngSkipHydration escapes for problematic components.

OnPush + signals

Apply OnPush everywhere and use signals for state. Angular skips unchanged subtrees — effectively free for components that haven't changed.

// LCP image with priority preload
@Component({ imports: [NgOptimizedImage] })
export class HeroComponent {}
// template:
// <img ngSrc="hero.jpg" width="1200" height="600" priority
//      sizes="(max-width: 768px) 100vw, 1200px" />

// Configure CDN loader for automatic format + resize
providers: [provideImgixLoader('https://cdn.example.com')]

// Lazy standalone component route
export const routes: Routes = [{
  path: 'report',
  loadComponent: () => import('./report.component').then(m => m.ReportComponent)
}];

// SSR-safe code (avoid browser-only APIs during server render)
@Component({...})
export class MapComponent {
  private platform = inject(PLATFORM_ID);
  showMap = isPlatformBrowser(this.platform);
}

State management — signals vs NgRx

Signal-based store (recommended for most apps)

@Injectable({ providedIn: 'root' })
export class CartStore {
  private readonly _items = signal<CartItem[]>([]);

  // Public read-only surface
  readonly items  = this._items.asReadonly();
  readonly count  = computed(() => this._items().length);
  readonly total  = computed(() => this._items().reduce((s, i) => s + i.price, 0));
  readonly isEmpty = computed(() => this._items().length === 0);

  add(item: CartItem)     { this._items.update(v => [...v, item]); }
  remove(id: string)      { this._items.update(v => v.filter(i => i.id !== id)); }
  clear()                 { this._items.set([]); }
  updateQty(id: string, qty: number) {
    this._items.update(v => v.map(i => i.id === id ? { ...i, qty } : i));
  }
}

Use signals + services for most apps. Add NgRx (or NgRx SignalStore) when you need Redux DevTools time-travel debugging, strict action audit trails, or team conventions already rely on it.

Signals

Signals · scenario
You convert a user-list HTTP call to toSignal() and the request fires twice on initial load. What went wrong and how do you fix it?
▼

Cause: toSignal(this.http.get(...)) subscribes to a cold observable. If the component mounts twice (e.g. inside a toggling @if), or if the service is not singleton, two HTTP calls fire. A second common cause is calling this.http.get() directly inside toSignal() — every injection of the service creates a new cold subscription.

Fix: Share the observable before converting with shareReplay(1).

@Injectable({ providedIn: 'root' })
export class UserService {
  private http = inject(HttpClient);

  // Cold observable shared so only ONE HTTP call fires
  private readonly users$ = this.http.get<User[]>('/api/users').pipe(shareReplay(1));
  readonly users = toSignal(this.users$, { initialValue: [] });
}

If the double-mount is the real cause, check whether the component hosting the signal is inside an @if that toggles on the same tick, or whether providedIn: 'root' is missing from the service.

Signals · scenario
A colleague uses effect() to sync a signal value into another signal. Code review says this is an anti-pattern. Why, and what is the correct pattern?
▼

Why it's wrong: Setting a signal inside effect() creates an indirect data dependency that is hard to trace and can cause infinite loops if not careful. Angular warns about this and may throw in strict mode.

// BAD — anti-pattern
effect(() => {
  this.fullName.set(`${this.firstName()} ${this.lastName()}`); // side-effect write
});

// GOOD — use computed() for derived values
fullName = computed(() => `${this.firstName()} ${this.lastName()}`);

effect() is for side effects outside Angular's reactivity system — DOM manipulation, logging, analytics, calling non-Angular APIs. Not for deriving new state.

If you genuinely need to write a signal from an effect (rare), use the allowSignalWrites: true option and document why computed() cannot solve it.

Change detection

Change detection · scenario
An OnPush component stops updating when you mutate an array. Diagnose it and fix it without removing OnPush.
▼

Diagnosis: OnPush only re-renders when an @Input reference changes. array.push(item) mutates the array but keeps the same reference — Angular sees an unchanged reference and skips the component.

Fixes (in order of preference):

1
Immutable update — create a new array reference: this.items = [...this.items, newItem]
2
Convert to signal — signals always notify regardless of change detection strategy
3
markForCheck() escape hatch — only if you must mutate and cannot use signals
// Option 2 — signal-based (preferred in Angular 16+)
items = signal<Item[]>([]);
addItem(i: Item) { this.items.update(v => [...v, i]); }
// template reads items() — automatically re-renders on change
Change detection · scenario
A developer adds a third-party library that updates UI outside Angular's zone. The changes don't show up. What is happening and what are the solutions?
▼

Cause: Zone.js intercepts browser APIs (setTimeout, Promise, XHR, etc.) to know when to run change detection. A third-party library that updates state using its own scheduler (requestAnimationFrame, custom event emitter) runs outside Zone.js and Angular never detects the change.

Solutions:

// Option 1: Run callback inside NgZone
constructor(private zone: NgZone) {
  thirdPartyLib.onUpdate((data) => {
    this.zone.run(() => { this.data = data; });
  });
}

// Option 2: Manually trigger change detection
constructor(private cdr: ChangeDetectorRef) {}
onLibUpdate(data: any) { this.data = data; this.cdr.detectChanges(); }

// Option 3 (best, Angular 16+): Store data in a signal — works outside zone
data = signal<any>(null);
onLibUpdate(d: any) { this.data.set(d); }  // Signal notifies, no zone needed

DI patterns

DI · scenario
You need a different ThemeService implementation on mobile vs desktop without if branching in every component. How?
▼ 
// 1. Define an abstract class as the interface
abstract class ThemeService { abstract theme: Signal<string>; }

// 2. Two implementations
@Injectable() class MobileThemeService extends ThemeService { theme = signal('compact'); }
@Injectable() class DesktopThemeService extends ThemeService { theme = signal('spacious'); }

// 3. Provide the correct one at the root
// main.ts
providers: [{
  provide: ThemeService,
  useClass: window.innerWidth < 768 ? MobileThemeService : DesktopThemeService
}]

// 4. Components just inject the abstract class — zero branching
@Component({...})
export class HeaderComponent {
  private theme = inject(ThemeService);
  currentTheme = this.theme.theme;  // Signal<string>
}

For runtime switching (user changes device orientation), use a single service with a computed() that derives theme from a device-size signal + user preference signal rather than two separate classes.

DI · scenario
After migrating to standalone, your lazy routes no longer get their scoped services. The services that were provided in lazy-loaded modules are now available globally. How do you fix scoping?
▼

Cause: Lazy-loaded modules created their own injector, scoping services to the module's lifetime. With standalone, there are no module injectors — all services default to the root injector unless you explicitly scope them.

// Solution: Route-level providers array
export const adminRoutes: Routes = [{
  path: '',
  // These providers are scoped to this route subtree
  providers: [
    AdminService,
    AuditLogService,
    { provide: AUTH_LEVEL, useValue: 'admin' }
  ],
  loadComponent: () => import('./admin-shell.component')
    .then(m => m.AdminShellComponent),
  children: [...]
}];

Services in providers: at route level live in the route's environment injector — scoped to the route subtree exactly like a lazy module used to scope them. They are destroyed when the user navigates away.

Performance

Performance · scenario
Your LCP is 4.2 seconds caused by a hero image. You are on Angular 16. Walk through the fix.
▼
1
Add NgOptimizedImage to the component imports
2
Replace <img src="..."> with <img ngSrc="..." priority> — priority injects <link rel="preload"> in <head> automatically
3
Add explicit width and height to prevent CLS
4
Configure a CDN loader for automatic WebP/AVIF format and resize-on-demand

@Component({ imports: [NgOptimizedImage], ... })

// template

Performance · scenario
A list component with 10,000 items causes noticeable scroll lag. OnPush is already applied. What else can you do?
▼

OnPush skips change detection but still renders all 10,000 DOM nodes. The DOM itself is the bottleneck.

1
Virtual scrolling (CDK) — renders only visible rows, recycles DOM elements as user scrolls
2
Pagination or infinite scroll — limit the list size at source
3
track by unique ID in @for — ensures Angular reuses DOM elements on re-render rather than recreating them
4
Defer non-critical columns — use @defer on complex cell content (charts, avatars) inside list rows
// CDK virtual scroll — renders only visible rows
<cdk-virtual-scroll-viewport itemSize="48" style="height: 600px">
  <div *cdkVirtualFor="let item of items; trackBy: trackId">
    {{ item.name }}
  </div>
</cdk-virtual-scroll-viewport>

Architecture

Architecture · scenario
You are migrating a large NgModule app to standalone. What is the safest path and what pitfalls should you watch for?
▼

Leaf-first migration — automated by the Angular schematic:

ng generate @angular/core:standalone
1
Run the schematic — it converts components leaf-first automatically
2
Validate each layer, commit, move up to parent components
3
Convert services last — providedIn: 'root' works with both NgModule and standalone
4
Replace AppModule with bootstrapApplication() once all feature modules are gone

Key pitfalls:

• SharedModules that export many components must be broken into sets of standalone components imported individually (or via barrel exports).

• forRoot() provider patterns must be converted to provide*() functions (e.g. provideRouter(), provideHttpClient()).

• Lazy module providers must migrate to route-level providers: arrays.

Architecture · scenario
Your team debates whether to use NgRx, Akita, or signals + services for state management in 2024. What do you recommend and why?
▼

Signals + services for most applications — less boilerplate, fully typed, Angular-native, zero additional dependency:

// Signal service — replaces NgRx for most use cases
@Injectable({ providedIn: 'root' })
export class OrderStore {
  private readonly _orders = signal<Order[]>([]);
  readonly orders  = this._orders.asReadonly();
  readonly pending = computed(() => this._orders().filter(o => o.status === 'pending'));
  readonly count   = computed(() => this._orders().length);

  loadOrders() { inject(OrderService).getAll$().subscribe(o => this._orders.set(o)); }
}

Add NgRx when: Redux DevTools time-travel is required, mutations arrive from many disconnected sources that need a strict action audit trail, or team conventions already rely on it.

NgRx SignalStore (NgRx 17+) bridges both worlds — signals under the hood, NgRx architecture on top. A strong choice for teams already invested in NgRx.

Avoid Akita/NGXS in new projects — they are not actively aligned with Angular's signal direction.

Testing

Testing · scenario
A component uses inject() in field initializers instead of constructor injection. Your existing test setup breaks with "inject() must be called from an injection context". How do you fix it?
▼

inject() in field initializers runs during TestBed.createComponent(). All dependencies must be provided in configureTestingModule before that call.

describe('UserComponent', () => {
  let httpSpy: jasmine.SpyObj<HttpClient>;

  beforeEach(async () => {
    httpSpy = jasmine.createSpyObj('HttpClient', ['get']);
    httpSpy.get.and.returnValue(of([]));

    await TestBed.configureTestingModule({
      imports: [UserComponent],         // standalone — import the component
      providers: [
        { provide: HttpClient, useValue: httpSpy },
        { provide: Router, useValue: { navigate: jasmine.createSpy() } }
      ]
    }).compileComponents();
  });

  it('should create', () => {
    const fixture = TestBed.createComponent(UserComponent);
    expect(fixture.componentInstance).toBeTruthy();
  });
});

// Testing signal stores in isolation — no TestBed needed
it('adds item to cart', () => {
  TestBed.runInInjectionContext(() => {
    const store = new CartStore();
    store.add({ id: '1', name: 'Book', price: 15 });
    expect(store.count()).toBe(1);
    expect(store.total()).toBe(15);
  });
});

Forms

Forms · scenario
Step 2 validators in a multi-step form need to read step 1 values (e.g. VAT required if EU country selected). How do you architect cross-step validation cleanly?
▼

Place the cross-field validator on the parent FormGroup rather than individual controls — it always has access to sibling values:

const checkoutForm = new FormGroup({
  billing:  new FormGroup({ country: new FormControl('', Validators.required) }),
  business: new FormGroup({ vatId:   new FormControl('') })
}, { validators: [vatRequiredForEU] });

function vatRequiredForEU(group: AbstractControl): ValidationErrors | null {
  const country = group.get('billing.country')?.value;
  const vatId   = group.get('business.vatId')?.value;
  const euCountries = ['DE', 'FR', 'NL', 'ES', 'IT'];
  if (euCountries.includes(country) && !vatId?.trim()) {
    return { vatRequired: { country } };
  }
  return null;
}

// In template — show error from parent group
@if (checkoutForm.errors?.['vatRequired']) {
  <p class="error">VAT ID required for EU businesses</p>
}

For async validation (e.g. validate VAT ID against an API), move the logic into a service that returns an Observable, inject it inside the validator factory using inject(), and return asyncValidators instead.

RxJS vs signals

RxJS vs signals · scenario
When should you keep RxJS in an Angular 17+ codebase versus converting to signals? Where is each appropriate?
▼ 
// Keep RxJS for:
// - HTTP with cancellation
readonly user$ = this.routeId$.pipe(switchMap(id => this.http.get(`/user/${id}`)));

// - Complex coordination
readonly dashboard$ = combineLatest([this.user$, this.permissions$]).pipe(
  debounceTime(10), map(([u, p]) => buildDashboard(u, p))
);

// - WebSocket streams
readonly messages$ = this.ws.connect('/chat').pipe(retry({ delay: 2000 }));

// Use signals for:
// - Local component state
selectedTab = signal<'overview' | 'details'>('overview');

// - Derived/computed values (replaces BehaviorSubject patterns)
isLoading   = signal(false);
errorMsg    = signal<string | null>(null);
displayName = computed(() => `${this.firstName()} ${this.lastName()}`);

// - @Input data flows — signal inputs natively
title = input.required<string>();

// Bridge when needed
readonly user = toSignal(this.user$, { initialValue: null });
readonly routeId$ = toObservable(this.routeIdSignal);

SSR & hydration

SSR · scenario
After enabling SSR in Angular 17, a date picker causes a hydration mismatch error: "Node not found". What is the cause and how do you fix it?
▼

Cause: Hydration mismatches occur when the server-rendered DOM differs from what the client renders on first paint. Common culprits for date pickers:

• new Date() called inline — different value on server (build time) vs client (user's local time)

• localStorage / window accessed during server render — throws ReferenceError

• Random IDs generated fresh each render

// Fix 1: Platform guard — skip browser-only UI during SSR
export class DatePickerComponent {
  private platform = inject(PLATFORM_ID);
  readonly isBrowser = isPlatformBrowser(this.platform);
}
// template: @if (isBrowser) { <date-picker /> }

// Fix 2: Stabilise via TransferState — share value server → client
import { TransferState, makeStateKey } from '@angular/core';
const TODAY_KEY = makeStateKey<string>('today');

export class AppComponent {
  private ts = inject(TransferState);
  readonly today = this.ts.get(TODAY_KEY, null) ??
    (() => { const d = new Date().toISOString(); this.ts.set(TODAY_KEY, d); return d; })();
}

// Fix 3: Last resort — skip hydration for this component only
<date-picker ngSkipHydration />

Router

Router · scenario
You want route params automatically bound to component inputs without writing ActivatedRoute subscriptions. How?
▼ 
// 1. Enable withComponentInputBinding() in providers
bootstrapApplication(AppComponent, {
  providers: [
    provideRouter(routes, withComponentInputBinding())
  ]
});

// 2. Route definition
{ path: 'user/:id', component: UserDetailComponent }

// 3. Component — params bound automatically to matching inputs
@Component({ standalone: true, ... })
export class UserDetailComponent {
  // :id param bound as string automatically
  id  = input.required<string>();
  // ?tab= query param bound automatically
  tab = input('overview');
  // Resolved route data also bound
  profile = input<UserProfile | null>(null);

  // Derive from bound signal
  user$ = toObservable(this.id).pipe(
    switchMap(id => this.userService.getUser$(id))
  );
}

No ActivatedRoute, no paramMap.pipe(map(...)), no unsubscription needed. The router manages binding lifecycle automatically.

Router · scenario
You need a route guard that injects a service and redirects based on a signal. How do you write a functional guard in Angular 15+?
▼ 
// Functional guard (v15+) — uses inject() internally
export const authGuard: CanActivateFn = (route, state) => {
  const auth   = inject(AuthService);
  const router = inject(Router);

  // Read signal value inside guard
  if (auth.isAuthenticated()) {
    return true;
  }
  // Redirect with return URL
  return router.createUrlTree(['/login'], {
    queryParams: { returnUrl: state.url }
  });
};

// Register on route
{ path: 'dashboard', canActivate: [authGuard], ... }

// Resolve guard — also functional
export const userResolver: ResolveFn<User> = (route) => {
  const id = route.paramMap.get('id')!;
  return inject(UserService).getUser$(id);
};
